# Web UI Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add an `openhancer ui` subcommand that launches a local web UI with real-time WebGPU preview, video scrubbing, preset management, and FFmpeg export.

**Architecture:** `Bun.serve` hosts a React SPA (bundled by Bun's bundler) and a minimal API (presets + export). The browser handles all preview rendering via WebGPU — video frames are read from a `<video>` element, processed through a chain of WGSL shaders matching the FFmpeg effect pipeline, and displayed on a `<canvas>`. The canvas renders at `min(source, 1080p)` with blur passes at half resolution. Export sends the preset to the server which runs the real FFmpeg pipeline on the original file.

**Tech Stack:** Bun (bundler + server), React 19, WebGPU (WGSL shaders), no other dependencies.

```ts
function listPresets(): string[] {
  const names: string[] = [];
  const builtinDir = join(import.meta.dir, "..", "..", "presets");
  const userDir = join(homedir(), ".openhancer", "presets");
  for (const dir of [builtinDir, userDir]) {
    if (existsSync(dir)) {
      for (const f of readdirSync(dir)) {
        if (f.endsWith(".json")) names.push(f.replace(".json", ""));
      }
    }
  }
  return [...new Set(names)];
}

export function createServer(port: number) {
  return Bun.serve({
    port,
    async fetch(req) {
      const url = new URL(req.url);

      if (url.pathname === "/api/schema") {
        return Response.json(EFFECT_SCHEMA);
      }

      if (url.pathname === "/api/presets" && req.method === "GET") {
        return Response.json(listPresets());
      }

      if (url.pathname === "/api/preset" && req.method === "GET") {
        const name = url.searchParams.get("name") || "default";
        try {
          return Response.json(loadPreset(name));
        } catch {
          return new Response("Preset not found", { status: 404 });
        }
      }

      if (url.pathname === "/api/presets" && req.method === "POST") {
        const body = await req.json();
        const { name, data } = body;
        if (!name || !data) return new Response("name and data required", { status: 400 });
        const userDir = join(homedir(), ".openhancer", "presets");
        if (!existsSync(userDir)) mkdirSync(userDir, { recursive: true });
        writeFileSync(join(userDir, `${name}.json`), JSON.stringify(data, null, 2));
        return Response.json({ ok: true });
      }

      if (url.pathname === "/api/export" && req.method === "POST") {
        // Accepts { filePath, params } — filePath is a local path the user provides
        // Returns SSE stream with progress, then download URL
        // Implemented in Task 9
        return new Response("Not implemented", { status: 501 });
      }

      // Static file serving (SPA) — implemented in Task 6
      const staticDir = join(import.meta.dir, "dist");
      const filePath = join(staticDir, url.pathname === "/" ? "index.html" : url.pathname);
      if (existsSync(filePath)) {
        return new Response(Bun.file(filePath));
      }
      const indexPath = join(staticDir, "index.html");
```

## Task 5: WebGPU Renderer — Core Pipeline

This is the heart of the UI. Build the WebGPU render pipeline that processes video/image frames through a chain of shader passes matching the FFmpeg effects.

**Architecture of the GPU pipeline:**

```
Source texture (video frame / image)
  → Pass 1: Color settings (fragment shader)
  → Pass 2: Halation — threshold highlights (fragment)
  → Pass 3: Halation — horizontal gaussian blur at half-res (fragment)
  → Pass 4: Halation — vertical gaussian blur at half-res (fragment)
  → Pass 5: Halation — screen blend with original (fragment)
  → Pass 6: Chromatic aberration (fragment — UV offset per channel)
  → Pass 7: Bloom — horizontal blur at half-res (fragment)
  → Pass 8: Bloom — vertical blur at half-res (fragment)
  → Pass 9: Bloom — screen blend (fragment)
  → Pass 10: Grain (fragment — procedural noise + overlay blend)
  → Pass 11: Vignette (fragment)
  → Pass 12: Split tone (fragment)
  → Pass 13: Camera shake (vertex transform — UV offset)
  → Output canvas
```

Passes that share logic (blur) use the same shader with different uniforms. Disabled effects are skipped (no GPU cost).

**Files:**
- Create: `src/ui/app/gpu/renderer.ts` — pipeline orchestrator
- Create: `src/ui/app/gpu/shaders.ts` — all WGSL shader source strings
- Create: `src/ui/app/gpu/passes.ts` — pass setup (bind groups, textures, uniforms)

```ts
// src/ui/app/gpu/shaders.ts

// Shared fullscreen quad vertex shader — used by all fragment passes
export const FULLSCREEN_VERT = /* wgsl */ `
struct VertexOutput {
  @builtin(position) position: vec4f,
  @location(0) uv: vec2f,
};

@vertex
fn vs(@builtin(vertex_index) i: u32) -> VertexOutput {
  // Fullscreen triangle trick: 3 vertices, no vertex buffer
  let uv = vec2f(f32((i << 1u) & 2u), f32(i & 2u));
  var out: VertexOutput;
  out.position = vec4f(uv * 2.0 - 1.0, 0.0, 1.0);
  out.uv = vec2f(uv.x, 1.0 - uv.y); // flip Y for texture coords
  return out;
}
`;

// --- Color Settings ---
// Matches: eq=contrast=C:brightness=B:saturation=S:gamma=G
// Plus: colortemperature, colorbalance(tint), bleach bypass
export const COLOR_SETTINGS_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct ColorParams {
  contrast: f32,       // options.contrast * (1 - fade)
  brightness: f32,     // exposure * 0.1 + fade * 0.05
  saturation: f32,     // subtractiveSat * richness
  gamma: f32,          // 1 - highlights * 0.5
  whiteBalance: f32,   // kelvin (6500 = neutral)
  tint: f32,           // -1 to 1 (green-magenta)
  bleachBypass: f32,   // 0 to 1
  _pad: f32,
};
@group(0) @binding(2) var<uniform> params: ColorParams;

fn rgb2luma(c: vec3f) -> f32 {
  return dot(c, vec3f(0.2126, 0.7152, 0.0722));
}

// Attempt to match FFmpeg colortemperature filter
fn applyWhiteBalance(color: vec3f, kelvin: f32) -> vec3f {
  if (abs(kelvin - 6500.0) < 1.0) { return color; }
  // Planckian locus approximation
  let t = kelvin / 100.0;
  var r: f32; var g: f32; var b: f32;
  if (t <= 66.0) {
    r = 1.0;
    g = clamp((0.39008 * log(t) - 0.63184), 0.0, 1.0);
  } else {
    r = clamp(1.292936 * pow(t - 60.0, -0.1332047592), 0.0, 1.0);
    g = clamp(1.129891 * pow(t - 60.0, -0.0755148492), 0.0, 1.0);
  }
  if (t >= 66.0) {
    b = 1.0;
  } else if (t <= 19.0) {
    b = 0.0;
  } else {
    b = clamp(0.54320 * log(t - 10.0) - 1.19625, 0.0, 1.0);
  }
  // Normalize relative to D65 (6500K)
  let d65 = vec3f(1.0, 0.9468, 0.9228);
  let wb = vec3f(r, g, b) / d65;
  return color * wb;
}

fn log(x: f32) -> f32 { return log2(x) / log2(2.718281828); }

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let color = textureSample(src, samp, uv).rgb;

  // Gamma (highlights)
  var c = pow(color, vec3f(params.gamma));

  // Contrast + brightness (eq filter)
  c = (c - 0.5) * params.contrast + 0.5 + params.brightness;
  c = clamp(c, vec3f(0.0), vec3f(1.0));

  // Saturation
  let luma = rgb2luma(c);
  c = mix(vec3f(luma), c, params.saturation);

  // White balance
  c = applyWhiteBalance(c, params.whiteBalance);

  // Tint (green-magenta on all tonal ranges)
  c.g = c.g + params.tint * 0.1;
  c = clamp(c, vec3f(0.0), vec3f(1.0));

  // Bleach bypass: blend with desaturated high-contrast version
  if (params.bleachBypass > 0.0) {
    let desat = vec3f(rgb2luma(c));
    let highContrast = (desat - 0.5) * 1.3 + 0.5;
    c = mix(c, clamp(highContrast, vec3f(0.0), vec3f(1.0)), params.bleachBypass);
  }

  return vec4f(c, 1.0);
}
`;

export const THRESHOLD_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct ThresholdParams {
  low: f32,   // 0.65
  high: f32,  // 0.75

  _pad2: f32,
};
@group(0) @binding(2) var<uniform> params: ThresholdParams;

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let c = textureSample(src, samp, uv).rgb;
  let luma = max(max(c.r, c.g), c.b);
  let t = smoothstep(params.low, params.high, luma);
  return vec4f(c * t, 1.0);
}
`;

// --- Gaussian blur (separable, used for halation + bloom) ---
export const BLUR_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct BlurParams {
  direction: vec2f,  // (1/w, 0) for horizontal, (0, 1/h) for vertical
  sigma: f32,
  _pad: f32,
};
@group(0) @binding(2) var<uniform> params: BlurParams;

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let sigma = max(params.sigma, 0.001);
  let radius = i32(ceil(sigma * 3.0));
  var color = vec3f(0.0);
  var weight_sum = 0.0;

  for (var i = -radius; i <= radius; i = i + 1) {
    let offset = params.direction * f32(i);
    let w = exp(-f32(i * i) / (2.0 * sigma * sigma));
    color += textureSample(src, samp, uv + offset).rgb * w;
    weight_sum += w;
  }

  return vec4f(color / weight_sum, 1.0);
}
`;

// --- Screen blend (halation + bloom compositing) ---
// Also handles hue rotation + saturation for halation glow
export const SCREEN_BLEND_FRAG = /* wgsl */ `
@group(0) @binding(0) var base_tex: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;
@group(0) @binding(2) var overlay_tex: texture_2d<f32>;

struct BlendParams {
  opacity: f32,
  hueShift: f32,   // 0-360 degrees (0 = no shift)
  saturation: f32,  // glow saturation multiplier
  _pad: f32,
};
@group(0) @binding(3) var<uniform> params: BlendParams;

fn rgb2hsv(c: vec3f) -> vec3f {
  let cmax = max(max(c.r, c.g), c.b);
  let cmin = min(min(c.r, c.g), c.b);
  let delta = cmax - cmin;
  var h: f32 = 0.0;
  if (delta > 0.001) {
    if (cmax == c.r) { h = ((c.g - c.b) / delta) % 6.0; }
    else if (cmax == c.g) { h = (c.b - c.r) / delta + 2.0; }
    else { h = (c.r - c.g) / delta + 4.0; }
    h = h / 6.0;
    if (h < 0.0) { h += 1.0; }
  }
  let s = select(0.0, delta / cmax, cmax > 0.0);
  return vec3f(h, s, cmax);
}

fn hsv2rgb(c: vec3f) -> vec3f {
  let h = c.x * 6.0;
  let s = c.y;
  let v = c.z;
  let i = floor(h);
  let f = h - i;
  let p = v * (1.0 - s);
  let q = v * (1.0 - s * f);
  let t = v * (1.0 - s * (1.0 - f));
  let idx = i32(i) % 6;
  if (idx == 0) { return vec3f(v, t, p); }
  if (idx == 1) { return vec3f(q, v, p); }
  if (idx == 2) { return vec3f(p, v, t); }
  if (idx == 3) { return vec3f(p, q, v); }
  if (idx == 4) { return vec3f(t, p, v); }
  return vec3f(v, p, q);
}

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let base = textureSample(base_tex, samp, uv).rgb;
  var overlay = textureSample(overlay_tex, samp, uv).rgb;

  // Apply hue shift + saturation to overlay (for halation glow color)
  if (params.hueShift != 0.0 || params.saturation != 1.0) {
    var hsv = rgb2hsv(overlay);
    hsv.x = fract(hsv.x + params.hueShift / 360.0);
    hsv.y = clamp(hsv.y * params.saturation, 0.0, 1.0);
    overlay = hsv2rgb(hsv);
  }

  // Screen blend: 1 - (1-a)(1-b)
  let blended = 1.0 - (1.0 - base) * (1.0 - overlay);
  return vec4f(mix(base, blended, params.opacity), 1.0);
}
`;

// --- Chromatic aberration ---
// Matches: scale R channel up, scale B channel down, recombine
export const ABERRATION_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct AberrationParams {
  offset: f32,  // amount * 0.02
  _pad1: f32,
  _pad2: f32,
  _pad3: f32,
};
@group(0) @binding(2) var<uniform> params: AberrationParams;

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let center = vec2f(0.5);
  let dir = uv - center;

  // R: zoom in (scale up = sample outward)
  let uv_r = center + dir * (1.0 + params.offset);
  // G: no shift
  // B: zoom out (scale down = sample inward)
  let uv_b = center + dir * (1.0 - params.offset);

  let r = textureSample(src, samp, uv_r).r;
  let g = textureSample(src, samp, uv).g;
  let b = textureSample(src, samp, uv_b).b;

  return vec4f(r, g, b, 1.0);
}
`;

// --- Grain ---
// Procedural noise, overlay blended, with defocus + saturation
export const GRAIN_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct GrainParams {
  amount: f32,
  size: f32,
  softness: f32,
  saturation: f32,
  imageDefocus: f32,
  time: f32,       // frame number for temporal variation
  texelSize: vec2f, // 1/width, 1/height
};
@group(0) @binding(2) var<uniform> params: GrainParams;

fn hash(p: vec2f) -> f32 {
  var p3 = fract(vec3f(p.xyx) * 0.1031);
  p3 += dot(p3, p3.yzx + 33.33);
  return fract((p3.x + p3.y) * p3.z);
}

fn grain_noise(uv: vec2f, t: f32) -> vec3f {
  let scale = max(1.0, params.size * 2.0 + 1.0);
  let coord = floor(uv * scale) + t;
  // Per-channel noise for colored grain
  let r = hash(coord + vec2f(0.0, 0.0)) * 2.0 - 1.0;
  let g = hash(coord + vec2f(1.7, 3.1)) * 2.0 - 1.0;
  let b = hash(coord + vec2f(5.3, 2.9)) * 2.0 - 1.0;
  let mono = (r + g + b) / 3.0;
  return mix(vec3f(mono), vec3f(r, g, b), params.saturation);
}

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  // Simple box-blur for defocus (sample 3x3 neighborhood)
  var color = vec3f(0.0);
  if (params.imageDefocus > 0.0) {
    let r = params.imageDefocus * 0.5;
    var total = 0.0;
    for (var x = -1; x <= 1; x++) {
      for (var y = -1; y <= 1; y++) {
        let off = vec2f(f32(x), f32(y)) * params.texelSize * r;
        color += textureSample(src, samp, uv + off).rgb;
        total += 1.0;
      }
    }
    color /= total;
  } else {
    color = textureSample(src, samp, uv).rgb;
  }

  // Generate grain
  let dims = vec2f(textureDimensions(src));
  let n = grain_noise(uv * dims, params.time);

  // Overlay blend
  let overlay = select(
    2.0 * color * (0.5 + n * 0.5),
    1.0 - 2.0 * (1.0 - color) * (0.5 - n * 0.5),
    color > vec3f(0.5)
  );

  return vec4f(mix(color, overlay, params.amount), 1.0);
}
`;

// --- Vignette ---
// Matches: vignette=angle=A:aspect=R
export const VIGNETTE_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct VignetteParams {
  angle: f32,   // amount * PI/2
  aspect: f32,  // 1 - size * 0.5
  _pad1: f32,
  _pad2: f32,
};
@group(0) @binding(2) var<uniform> params: VignetteParams;

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let color = textureSample(src, samp, uv).rgb;
  let center = uv - 0.5;
  let adjusted = vec2f(center.x, center.y * params.aspect);
  let dist = length(adjusted) * 2.0;
  let vig = cos(min(dist * params.angle, 3.14159265 * 0.5));
  return vec4f(color * vig * vig, 1.0);
}
`;

// --- Split Tone ---
// Matches: colorbalance with computed shadow/highlight/mid tints
export const SPLIT_TONE_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct SplitToneParams {
  shadowR: f32,
  shadowB: f32,
  highlightR: f32,
  highlightB: f32,
  midR: f32,
  amount: f32,
  protectNeutrals: f32, // 0 or 1
  _pad: f32,
};
@group(0) @binding(2) var<uniform> params: SplitToneParams;

fn rgb2luma(c: vec3f) -> f32 {
  return dot(c, vec3f(0.2126, 0.7152, 0.0722));
}

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let color = textureSample(src, samp, uv).rgb;
  let luma = rgb2luma(color);

  // Shadow tint (applied to darks)
  let shadow_weight = 1.0 - smoothstep(0.0, 0.5, luma);
  // Highlight tint (applied to brights)
  let highlight_weight = smoothstep(0.5, 1.0, luma);

  var toned = color;
  toned.r += params.shadowR * shadow_weight + params.highlightR * highlight_weight + params.midR;
  toned.b += params.shadowB * shadow_weight + params.highlightB * highlight_weight;
  toned = clamp(toned, vec3f(0.0), vec3f(1.0));

  if (params.protectNeutrals > 0.5) {
    toned = mix(color, toned, params.amount);
  }

  return vec4f(toned, 1.0);
}
`;

// --- Camera Shake ---
// UV offset based on sin(frame/period) — matches FFmpeg crop expression
export const CAMERA_SHAKE_FRAG = /* wgsl */ `
@group(0) @binding(0) var src: texture_2d<f32>;
@group(0) @binding(1) var samp: sampler;

struct ShakeParams {
  amplitude: f32,   // amount * 3 pixels, normalized to UV
  period1: f32,
  period2: f32,
  frame: f32,
};
@group(0) @binding(2) var<uniform> params: ShakeParams;

@fragment
fn fs(@location(0) uv: vec2f) -> @location(0) vec4f {
  let dx = params.amplitude * sin(params.frame / params.period1);
  let dy = params.amplitude * sin(params.frame / params.period2 + 1.3);
  let shifted = uv + vec2f(dx, dy);
  return textureSample(src, samp, clamp(shifted, vec2f(0.0), vec2f(1.0)));
}
`;
```

```ts
// src/ui/app/gpu/passes.ts

export interface RenderPass {
  pipeline: GPURenderPipeline;
  bindGroup: GPUBindGroup;
  uniformBuffer: GPUBuffer;
}

export function createFullscreenPipeline(
  device: GPUDevice,
  vertexShader: string,
  fragmentShader: string,
  bindGroupLayout: GPUBindGroupLayout,
  format: GPUTextureFormat,
): GPURenderPipeline {
  return device.createRenderPipeline({
    layout: device.createPipelineLayout({ bindGroupLayouts: [bindGroupLayout] }),
    vertex: {
      module: device.createShaderModule({ code: vertexShader }),
      entryPoint: "vs",
    },
    fragment: {
      module: device.createShaderModule({ code: fragmentShader }),
      entryPoint: "fs",
      targets: [{ format }],
    },
    primitive: { topology: "triangle-list" },
  });
}

export function createTexture(device: GPUDevice, width: number, height: number, format: GPUTextureFormat): GPUTexture {
  return device.createTexture({
    size: { width, height },
    format,
    usage: GPUTextureUsage.TEXTURE_BINDING | GPUTextureUsage.RENDER_ATTACHMENT | GPUTextureUsage.COPY_DST,
  });
}

export function runPass(
  encoder: GPUCommandEncoder,
  pipeline: GPURenderPipeline,
  bindGroup: GPUBindGroup,
  target: GPUTextureView,
) {
  const pass = encoder.beginRenderPass({
    colorAttachments: [{
      view: target,
      loadOp: "clear",
      storeOp: "store",
      clearValue: { r: 0, g: 0, b: 0, a: 1 },
    }],
  });
  pass.setPipeline(pipeline);
  pass.setBindGroup(0, bindGroup);
  pass.draw(3); // fullscreen triangle
  pass.end();
}
```

## Summary

| Task | What | Key Deliverable |
|------|------|-----------------|
| 1 | Option schema | `src/schema.ts` — single source of truth |
| 2 | Subcommand routing | `openhancer ui` detection |
| 3 | API server | Presets + export endpoints (no preview endpoint) |
| 4 | React scaffold | Bun-bundled SPA |
| 5 | WebGPU renderer | WGSL shaders matching all FFmpeg effects, render pipeline |
| 6 | Upload + video player | Drag-and-drop, `<video>` + WebGPU canvas, scrubber |
| 7 | Controls panel | Schema-driven sliders → instant GPU uniform updates |
| 8 | Export | Server-side FFmpeg render + SSE progress + download |
| 9 | Build integration | Single binary with embedded UI |
| 10 | API e2e tests | Server endpoint verification (bun:test) |
| 11 | Playwright setup | Config, helpers, Chromium install |
| 12 | Playwright e2e tests | Full UI flow: load, upload, controls, presets, export |

**Key architectural difference from original plan:** The browser does ALL preview rendering via WebGPU. The server is only needed for preset CRUD and final FFmpeg export. This means:
- No `/api/preview` endpoint, no temp files, no debounce latency
- 60fps real-time preview as sliders move
- Canvas renders at `min(source, 1080p)`, blur passes at half-res
- Video scrubbing is instant (no server round-trip)
- The file never leaves the browser until export time
